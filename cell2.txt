# --- 3. Helper Functions ---
def get_latest_map():
    files = glob.glob(os.path.join(MAPPING_DIR, "*.csv"))
    return max(files, key=os.path.getmtime) if files else None

def on_list_file_change(c):
    fname = up_list.value[0]['name']
    if "User Listing" not in fname:
        txt_list.value = "<b style='color:red;'>❌ Filename must contain 'User Listing'</b>"
    else:
        txt_list.value = f"<b style='color:green;'>✅ Selected: {fname}</b>"
up_list.observe(on_list_file_change, 'value')

map_path = get_latest_map()
txt_map = widgets.HTML(value=f"<b style='color:green;'>✅ Default: {os.path.basename(map_path)}</b>" if map_path else "<i>No map found</i>", layout=widgets.Layout(margin='0 10px'))

def smart_detect_column(df, candidates, avoid_terms=[]):
    """
    Tries to find the best column match.
    1. Look for exact matches.
    2. Look for partial matches (but avoid terms like 'Manager' if looking for 'Dept').
    """
    cols = [str(c).strip() for c in df.columns]
    cols_lower = [c.lower() for c in cols]
    
    # 1. Exact Match Strategy
    for cand in candidates:
        if cand in cols_lower:
            return df.columns[cols_lower.index(cand)]
            
    # 2. Contains Strategy (Strict)
    for cand in candidates:
        for i, c_low in enumerate(cols_lower):
            if cand in c_low:
                # If we find "Department", make sure it's not "Department Head" if we can help it
                if avoid_terms and any(bad in c_low for bad in avoid_terms):
                    continue
                return df.columns[i]
    return None

# --- 4. Logic: Analysis ---
def do_process(b):
    global processed_df, current_fname, review_registry, mapping_data
    out_area.clear_output()
    review_registry = []
    
    if not up_list.value or "User Listing" not in up_list.value[0]['name']:
        with out_area: print("Error: Valid 'User Listing' file required.")
        return
    
    b.disabled = True
    try:
        logger.info("--- Starting Analysis ---")
        
        # A. Load Map & Parse Branches
        m_src = io.BytesIO(up_map.value[0]['content']) if up_map.value else map_path
        if not m_src: raise Exception("No Mapping CSV found!")
            
        df_m = pd.read_csv(m_src)
        
        # Improved Column Detection
        col_m_email = smart_detect_column(df_m, ["email address", "email", "mail"], avoid_terms=["manager", "reviewer"])
        col_m_dept = smart_detect_column(df_m, ["department", "dept", "function"], avoid_terms=["head", "manager", "lead"])
        col_m_rev = smart_detect_column(df_m, ["reviewer", "owner", "manager", "name"])
        
        if not col_m_rev or not col_m_dept:
            with out_area: print(f"Error: Could not identify columns. Found: {list(df_m.columns)}")
            return

        logger.info(f"Mapping Cols Detected -> Dept: '{col_m_dept}' | Reviewer: '{col_m_rev}'")

        # Build Lookups (Mimic v5.3: astype(str).str.lower().str.strip())
        mapping_data['emails'] = {}
        if col_m_email:
            mapping_data['emails'] = dict(zip(df_m[col_m_email].astype(str).str.lower().str.strip(), df_m[col_m_rev]))
            
        # Dept Lookup
        mapping_data['depts'] = dict(zip(df_m[col_m_dept].astype(str).str.lower().str.strip(), df_m[col_m_rev]))
        
        # UI Lists
        raw_depts = sorted([str(d).strip() for d in df_m[col_m_dept].unique()])
        mapping_data['all_branches'] = [("Select Branch...", "")] + [(d, d.lower()) for d in raw_depts if "branch" in d.lower()]
        mapping_data['all_depts'] = [("Select Dept...", "")] + [(d, d.lower()) for d in raw_depts if "branch" not in d.lower()]

        # B. Load User List
        f_item = up_list.value[0]
        current_fname = f_item['name']
        df_u = pd.read_csv(io.BytesIO(f_item['content'])) if current_fname.endswith('.csv') else pd.read_excel(io.BytesIO(f_item['content']))
        
        col_email = smart_detect_column(df_u, ["email", "mail", "user principal", "upn"])
        if not col_email: col_email = df_u.columns[0]
        
        # C. AD Fetch
        emails = df_u[col_email].dropna().unique().tolist()
        with out_area: print(f"Syncing {len(emails)} emails with AD...")
        with ThreadPoolExecutor(max_workers=25) as ex:
            ad_results = {r['email']: r for r in list(ex.map(fetch_ad, emails))}

        # D. Processing & Matching
        final_rows = []
        widget_rows = []
        
        # Header for Grid
        header_box = widgets.HBox([
            widgets.Label("Remove", layout=widgets.Layout(width='60px', font_weight='bold')),
            widgets.Label("User Email", layout=widgets.Layout(flex='2', font_weight='bold')),
            widgets.Label("AD Status & Info", layout=widgets.Layout(flex='2', font_weight='bold')),
            widgets.Label("Branch?", layout=widgets.Layout(width='60px', font_weight='bold')),
            widgets.Label("Select Mapping", layout=widgets.Layout(flex='2', font_weight='bold')),
            widgets.Label("Result Reviewer", layout=widgets.Layout(flex='1', font_weight='bold')),
        ], layout=widgets.Layout(border='0px 0px 2px 0px solid gray', padding='5px'))

        debug_logs = []

        for idx, row in df_u.iterrows():
            raw_email = str(row[col_email]).strip().lower()
            ad = ad_results.get(raw_email, {"status": "Not Found", "dept": "N/A", "active": "N/A", "last_login": None})
            
            # === LOGIC FROM v5.3 RESTORED ===
            # 1. Clean the AD Department string exactly like v5.3
            ad_dept_raw = str(ad['dept'])
            if " - " in ad_dept_raw:
                clean_dept = ad_dept_raw.split(" - ")[-1].strip().lower()
            else:
                clean_dept = ad_dept_raw.strip().lower()
                
            # 2. Matching Priority
            # First try Email Map
            reviewer = mapping_data['emails'].get(raw_email)
            
            # Second try Dept Map (Cleaned)
            if not reviewer:
                reviewer = mapping_data['depts'].get(clean_dept)
                
            # Third try Dept Map (Raw) - Just in case map has "001 - IT"
            if not reviewer:
                 reviewer = mapping_data['depts'].get(ad_dept_raw.strip().lower())
            
            # === END RESTORED LOGIC ===

            is_problem = not reviewer or str(reviewer).strip().lower() in ["nan", "none", "", "()", "null"]
            
            if is_problem:
                if len(debug_logs) < 3: debug_logs.append(f"Fail: {raw_email} | AD Dept: {ad['dept']} | Clean: {clean_dept}")
                reviewer = "(please manual review)"
                
                # Colors & Widgets
                is_active = ad['active'] is True
                found = ad['status'] == "Found"
                remove_default = False
                
                if is_active:
                    status_html = f"<b style='color:green;'>Act: True</b><br><span style='font-size:10px; color:gray'>{ad['dept']}</span>"
                elif found and not is_active:
                    last_seen = ad.get('last_login') or "Unknown"
                    status_html = f"<span style='color:#b8860b;'>Inactive (Last: {last_seen})</span>"
                    remove_default = True
                else:
                    status_html = f"<b style='color:red;'>Act: N/A (No AD Record)</b>"
                    remove_default = True

                chk = widgets.Checkbox(value=remove_default, indent=False, layout=widgets.Layout(width='40px'))
                email_lbl = widgets.HTML(value=f"<div style='font-size:11px'>{raw_email}</div>")
                stat_lbl = widgets.HTML(value=f"<div style='font-size:11px; line-height:1.1'>{status_html}</div>")
                
                tgl = widgets.ToggleButton(value=False, description='Br', tooltip='Toggle Branch List', 
                                           button_style='', layout=widgets.Layout(width='40px', height='25px'))
                
                drp = widgets.Dropdown(options=mapping_data['all_depts'], layout=widgets.Layout(width='95%'))
                res_lbl = widgets.Label(value="Unassigned")

                def on_tgl_change(change, d=drp):
                    d.options = mapping_data['all_branches'] if change['new'] else mapping_data['all_depts']
                    d.value = ""
                
                def on_drp_change(change, l=res_lbl): 
                    l.value = str(mapping_data['depts'].get(change['new'], "Manual Selection"))

                tgl.observe(on_tgl_change, names='value')
                drp.observe(on_drp_change, names='value')

                review_registry.append({'index': idx, 'chk': chk, 'drp': drp, 'lbl': res_lbl, 'email': raw_email})
                
                row_box = widgets.HBox([
                    chk, 
                    widgets.Box([email_lbl], layout=widgets.Layout(flex='2')),
                    widgets.Box([stat_lbl], layout=widgets.Layout(flex='2')),
                    tgl,
                    widgets.Box([drp], layout=widgets.Layout(flex='2')),
                    widgets.Box([res_lbl], layout=widgets.Layout(flex='1'))
                ], layout=widgets.Layout(border='0px 0px 1px 0px solid #eee', padding='4px 0'))
                
                widget_rows.append(row_box)

            new_row = row.to_dict()
            new_row.update({"Reviewer": reviewer, "Reviewer's Response": "", "Details of Access Changes": ""})
            final_rows.append(new_row)

        processed_df = pd.DataFrame(final_rows)
        btn_save.disabled = False
        
        with out_area:
            print(f"Analysis Complete. Total: {len(processed_df)} | Manual Review: {len(widget_rows)}")
            if debug_logs:
                print("--- DEBUG: Sample Mapping Failures ---")
                for l in debug_logs: print(l)
            
            if widget_rows:
                display(widgets.VBox([header_box] + widget_rows, layout=widgets.Layout(width='100%')))
            else:
                display(HTML("<h4 style='color:green'>All records matched automatically!</h4>"))
                
    except Exception as e:
        with out_area: print(f"Error: {str(e)}")
        logger.error("Process Error", exc_info=True)
    finally: b.disabled = False

# --- 5. Logic: Save ---
def do_save(b):
    global processed_df
    b.disabled = True
    try:
        df_final = processed_df.copy()
        to_drop = []
        logs_txt = []
        
        for item in review_registry:
            if item['chk'].value: 
                to_drop.append(item['index'])
                logs_txt.append(f"Dropped: {item['email']}")
            elif item['drp'].value != "": 
                df_final.at[item['index'], 'Reviewer'] = item['lbl'].value
                logs_txt.append(f"Assigned: {item['email']} -> {item['lbl'].value}")
        
        df_final = df_final.drop(to_drop)
        fpath = os.path.join(BASE_DIR, current_fname.split('.')[0] + "_Reviewed.xlsx")
        df_final.to_excel(fpath, index=False)
        
        wb = load_workbook(fpath)
        ws = wb.active
        col_resp = next((i for i, c in enumerate(ws[1], 1) if c.value == "Reviewer's Response"), None)
        if col_resp:
            dv = DataValidation(type="list", formula1='"Approved,Denied,Changes Required"', allow_blank=True)
            dv.errorTitle, dv.error = "Error", "Please select valid option"
            letter = get_column_letter(col_resp)
            dv.add(f"{letter}2:{letter}{len(df_final)+1}")
            ws.add_data_validation(dv)
        wb.save(fpath)
        
        logger.info(f"Saved to {fpath}. Dropped: {len(to_drop)}. Assigned: {len(logs_txt)-len(to_drop)}")
        with out_area: 
            display(HTML(f"<b style='color:blue;'>✔ File Saved: {fpath}</b>"))
            print(f"Rows Removed: {len(to_drop)}")

    except Exception as e:
        with out_area: print(f"Save Error: {str(e)}")
        logger.error("Save Error", exc_info=True)
    finally: b.disabled = False

# --- 6. Final UI Assembly ---
btn_process.on_click(do_process)
btn_save.on_click(do_save)

ui = widgets.VBox([
    widgets.HTML("<h4>🛡️ AER Generator v5.10 (Fix)</h4>"), 
    widgets.HBox([up_list, txt_list]), 
    widgets.HBox([up_map, txt_map]), 
    widgets.HBox([btn_process, btn_save], layout=widgets.Layout(margin='15px 0')), 
    out_area
])

clear_output()
display(ui)
