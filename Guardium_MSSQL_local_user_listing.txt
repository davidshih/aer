sql
SELECT COUNT(*)
FROM sys.server_principals
WHERE
    type = 'S'
    OR (
        type IN ('U','G')
        AND name LIKE CAST(SERVERPROPERTY('MachineName') AS nvarchar(128)) + '\%'
    );
C) SQL Statement for Detail（回傳「一欄、多列」字串清單）
重點：Detail 查詢要「每列只回傳一個字串欄位」，Guardium 才能把它組成 detail string。

sql
SELECT
    name + ' | ' + type_desc + ' | ' +
    CASE WHEN is_disabled = 1 THEN 'Disabled' ELSE 'Enabled' END
FROM sys.server_principals
WHERE
    type = 'S'
    OR (
        type IN ('U','G')
        AND name LIKE CAST(SERVERPROPERTY('MachineName') AS nvarchar(128)) + '\%'
    )
ORDER BY type_desc, name;
