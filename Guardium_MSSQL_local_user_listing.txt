SELECT
    name AS Account_Name,
    type_desc AS Account_Type,
    CASE is_disabled
        WHEN 1 THEN 'Disabled'
        WHEN 0 THEN 'Enabled'
    END AS Status
FROM
    sys.server_principals
WHERE

    type = 'S'

    OR (
        type IN ('U', 'G') -- Windows User or Group
        AND (
            name LIKE CAST(SERVERPROPERTY('MachineName') AS NVARCHAR) + '\%'
            OR name LIKE 'NT AUTHORITY\%'
            OR name LIKE 'BUILTIN\%'
        )
    )
ORDER BY type_desc, name;
