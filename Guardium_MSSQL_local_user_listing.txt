) SQL statement（回傳數字）
（回傳 1 個整數）

sql
SELECT COUNT(*)
FROM sys.server_principals
WHERE
    type = 'S'
    OR (
        type IN ('U','G')
        AND (
            name LIKE CAST(SERVERPROPERTY('MachineName') AS NVARCHAR(128)) + '\%'
        )
    );
2) SQL Statement for Detail（回傳「一欄、多列」字串）
（每列一個字串，Guardium 會組成 detail string）
​

sql
SELECT
    CONCAT(
        name, ' | ', type_desc, ' | ',
        CASE is_disabled WHEN 1 THEN 'Disabled' ELSE 'Enabled' END
    ) AS detail_item
FROM sys.server_principals
WHERE
    type = 'S'
    OR (
        type IN ('U','G')
        AND (name LIKE CAST(SERVERPROPERTY('MachineName') AS NVARCHAR(128)) + '\%')
    )
ORDER BY type_desc, name;
